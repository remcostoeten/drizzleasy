---
title: Drizzleasy
description: Ultra-simple, type-safe CRUD operations for Next.js with Drizzle ORM.
---

Drizzleasy makes CRUD operations and database management ridiculously easy. 100% type-safe with chainable syntax and clean DX.

```bash
npm install @remcostoeten/drizzleasy
```

```ts
// Quickstart
import { initializeConnection, readFn, createFn } from '@remcostoeten/drizzleasy'

type TUser = { id: string; name: string; email: string; status: 'active' | 'inactive' }

async function main() {
  const db = await initializeConnection(process.env.DATABASE_URL!)
  const read = readFn<TUser>()
  const create = createFn<TUser>()

  await create('users')({ name: 'Jane', email: 'jane@example.com', status: 'active' })
  const { data: users } = await read('users')()
}
```
